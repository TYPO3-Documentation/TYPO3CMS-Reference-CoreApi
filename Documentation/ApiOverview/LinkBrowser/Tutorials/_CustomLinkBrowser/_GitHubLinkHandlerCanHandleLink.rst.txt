   class GitHubLinkHandler implements LinkHandlerInterface
   {
       public function canHandleLink(array $linkParts): bool
       {
           if (!isset($linkParts['type']) || $linkParts['type'] !== 'github') {
               return false;
           }
           $this->linkParts = $linkParts;
           return true;
       }
   }
